\section{L4 Naive Bayes + Embeddings}
\subsection*{Multinomial Naive Bayes}
MAP: $\hat c=\arg\max_{c\in C}P(c\mid x)=\arg\max_{c\in C}P(c)P(x\mid c)$.\\
Likelihood for bag-of-words counts $x_i$: $P(x\mid c)=\dfrac{n!}{\prod_{i=1}^{|V|}x_i!}\prod_{i=1}^{|V|}\theta_{c,i}^{x_i}$, $n=\sum_i x_i$.\\
Log-score: $\hat c=\arg\max_c\left[\log P(c)+\sum_{i=1}^{|V|}x_i\log\theta_{c,i}\right]$.\\
Smoothing: $\theta_{c,i}=\dfrac{n_{i,c}+\alpha}{n_c+\alpha|V|}$ (MLE is $\alpha=0$).

\subsection*{Word Vectors}
Dot product: $a\cdot b=\sum_i a_ib_i$; norm: $\|a\|=\sqrt{\sum_i a_i^2}$; cosine: $\cos(a,b)=\dfrac{a\cdot b}{\|a\|\|b\|}$.\\
Small context window tends to syntactic neighbors; larger window tends to topical neighbors.

\subsection*{Analogy Rule}
For $a:a^*::b:b^*$, $v(b^*)\approx v(b)-v(a)+v(a^*)$.\\
Works better for some relation types than others (frequency and semantic stability matter).

\subsection*{SGNS Core}
$P(+\mid w,c)=\sigma(u_c\cdot v_w)$, $P(-\mid w,c)=\sigma(-u_c\cdot v_w)$.\\
One positive + $k$ negatives CE term: $L=-\left[\log\sigma(u_{c^+}\cdot v_w)+\sum_{i=1}^{k}\log\sigma(-u_{c_i^-}\cdot v_w)\right]$.\\
Useful derivatives: $\dfrac{d}{d\alpha}\log\sigma(\alpha)=\sigma(-\alpha)$, $\dfrac{d}{d\alpha}\log\sigma(-\alpha)=-\sigma(\alpha)$.
