\section{L4: Naive Bayes, Vectors, SGNS, A1 Repro}
\subsection*{Multinomial Naive Bayes}
MAP classifier:
\[
\hat c_{MAP}=\arg\max_{c\in C}P(c\mid x).
\]
\[
\hat c_{MAP}=\arg\max_{c\in C}P(c)P(x\mid c).
\]
Multinomial likelihood for bag-of-words counts $x_i$:
\[
P(x\mid c)=\frac{n!}{\prod_{i=1}^{|V|}x_i!}
\prod_{i=1}^{|V|}\theta_{c,i}^{x_i},
\]
\[
n=\sum_i x_i.
\]
Log-space decision rule:
\[
\hat c=\arg\max_{c\in C}
\left[\log P(c)+\sum_{i=1}^{|V|}x_i\log\theta_{c,i}\right].
\]
MLE and add-$\alpha$ smoothing:
\[
\theta_{c,i}^{MLE}=\frac{n_{i,c}}{n_c},
\qquad
\theta_{c,i}=\frac{n_{i,c}+\alpha}{n_c+\alpha|V|}.
\]

\subsection*{Word Vectors and Similarity}
Dot product:
\[
a\cdot b=\sum_i a_i b_i.
\]
Norm and cosine:
\[
\|a\|=\sqrt{\sum_i a_i^2},
\qquad
\cos(a,b)=\frac{a\cdot b}{\|a\|\|b\|}.
\]
Use cosine when vector length/frequency bias is undesirable.

\subsection*{SGNS Core (lecture form)}
For target word $w$ and context word $c$:
\[
P(+\mid w,c)=\sigma(u_c\cdot v_w).
\]
\[
P(-\mid w,c)=\sigma(-u_c\cdot v_w).
\]
One positive + $k$ negatives objective:
\[
\begin{array}{l}
L_{CE}= -\Big[
\log\sigma(u_{c^+}\cdot v_w)\\
\quad +\sum_{i=1}^{k}\log\sigma(-u_{c_i^-}\cdot v_w)
\Big].
\end{array}
\]
Useful derivatives:
\[
\frac{d}{d\alpha}\log\sigma(\alpha)=\sigma(-\alpha).
\]
\[
\frac{d}{d\alpha}\log\sigma(-\alpha)=-\sigma(\alpha).
\]

\subsection*{A1 Repro (exam-speed chain)}
\textbf{Q1 MCQ key:} (a) C, (b) A, (c) B, (d) C, (e) C.\\
\textbf{Q2(a)} add-one bigram:
\[
P(Sam\mid am)=
\frac{C(am,Sam)+1}{C(am)+|V|}
\]
\[
P(Sam\mid am)=\frac{2+1}{3+11}=\frac{3}{14}.
\]
\textbf{Q2(b)} interpolation ($\lambda_1=\lambda_2=\frac{1}{2}$):
\[
P(Sam\mid am)=
\frac{1}{2}\cdot\frac{2}{3}
+\frac{1}{2}\cdot\frac{4}{25}
=\frac{31}{75}.
\]
\textbf{Q3} No EOS modeling:
\[
\sum_{w_{1:\ell}}P(w_{1:\ell})=1\ \text{for each fixed }\ell,
\]
not a valid global distribution across all lengths.\\
\textbf{Q4} NB derivation path: Bayes $\rightarrow$ drop class-independent terms $\rightarrow$ log-linear score.\\
\textbf{Q5} Logistic objective and gradient: same CE and gradient forms as L3.

\subsection*{L4 Pitfalls}
- Compute NB in log space to avoid underflow.\\
- Without smoothing, one unseen word can zero a class score.\\
- Dot product alone can over-favor frequent words; cosine is safer.
