\section{L3: Logistic Regression, Optimization, Evaluation}
\subsection*{Model and Decision Rule}
Binary logistic regression:\\
\[
\hat y = \sigma(w\cdot x+b),\qquad
\sigma(z)=\frac{1}{1+e^{-z}}.
\]
\[
p(y=1\mid x)=\hat y,\qquad p(y=0\mid x)=1-\hat y.
\]
Classifier with threshold $0.5$:\\
\[
\hat y_{cls}=\begin{cases}
1,&\sigma(w\cdot x+b)\ge 0.5\\
0,&\text{otherwise}
\end{cases}.
\]
Decision boundary: $w\cdot x+b=0$.\\

\subsection*{Cross-Entropy / NLL}
Single example:\\
\[
L_{CE}(\hat y,y)=-\big[y\log\hat y+(1-y)\log(1-\hat y)\big].
\]
Dataset objective:\\
\[
J(w,b)=\sum_{i=1}^{m}L_{CE}(\hat y^{(i)},y^{(i)}).
\]

\subsection*{Gradient and SGD}
For one example (textbook Eq. form):\\
\[
\frac{\partial L_{CE}}{\partial w_j}=\big(\sigma(w\cdot x+b)-y\big)x_j.
\]
\[
\frac{\partial L_{CE}}{\partial b}=\sigma(w\cdot x+b)-y.
\]
SGD update:\\
\[
w_j\leftarrow w_j-\eta\frac{\partial L}{\partial w_j}.
\]
\[
b\leftarrow b-\eta\frac{\partial L}{\partial b}.
\]
Vector form: $\theta\leftarrow\theta-\eta\nabla_\theta L$.

\subsection*{Confusion-Matrix Metrics}
\[
\text{Accuracy}=\frac{TP+TN}{TP+TN+FP+FN}.
\]
\[
\text{Precision}=\frac{TP}{TP+FP}.
\]
\[
\text{Recall}=\frac{TP}{TP+FN}.
\]
\[
F_1=\frac{2\,\text{Precision}\cdot\text{Recall}}{\text{Precision}+\text{Recall}}.
\]
Multi-class aggregation:\\
- Macro: average class-wise metrics equally.\\
- Micro: aggregate counts first, then compute metric.

\subsection*{L3 Pitfalls}
- Accuracy can be misleading under class imbalance; report precision/recall/$F_1$.\\
- Learning rate too high overshoots; too low converges very slowly.\\
- Feature scaling/representation quality strongly affects convergence and boundary quality.
