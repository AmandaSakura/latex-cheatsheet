\section{L3: Logistic Regression, Optimization, Evaluation}
\subsection*{Binary Logistic Regression (Compact)}
Setup: examples $(x^{(i)},y^{(i)})$, $x=[x_1,\ldots,x_n]$, weights $w=[w_1,\ldots,w_n]$, $y\in\{0,1\}$.\\
\[
\begin{aligned}
z&=w\cdot x+b=\sum_{j=1}^{n}w_jx_j+b,\\
\hat y&=\sigma(z)=\frac{1}{1+\exp(-z)}=\sigma(w\cdot x+b),\\
p(y=1\mid x)&=\hat y,\quad p(y=0\mid x)=1-\hat y=\sigma(-(w\cdot x+b)).
\end{aligned}
\]
Useful identity: $1-\sigma(a)=\sigma(-a)$. Decision boundary: $w\cdot x+b=0$.

\subsection*{Training Objective (CE/NLL)}
Conditional Bernoulli: $p(y\mid x)=\hat y^{\,y}(1-\hat y)^{1-y}$.\\
\[
L_{CE}(\hat y,y)=-\left[y\log\hat y+(1-y)\log(1-\hat y)\right].
\]
Dataset objective: $J(w,b)=\sum_{i=1}^{m}L_{CE}^{(i)}$, or $\bar J=\dfrac{1}{m}\sum_{i=1}^{m}L_{CE}^{(i)}$. Equivalent view: $L_{CE}=-\log p(y\mid x;w,b)$.

\subsection*{Gradient and SGD}
\[
\frac{\partial L_{CE}}{\partial w_j}=(\hat y-y)x_j,\qquad \frac{\partial L_{CE}}{\partial b}=\hat y-y,\qquad \nabla_wL=(\hat y-y)x.
\]
SGD: $w\leftarrow w-\eta\nabla_wL$, $b\leftarrow b-\eta\dfrac{\partial L}{\partial b}$, equivalently $\theta_{t+1}=\theta_t-\eta\nabla_\theta L$.

\subsection*{Evaluation + Confusion Matrix}
Confusion matrix mapping: $TP$ (gold+, pred+), $FN$ (gold+, pred-), $FP$ (gold-, pred+), $TN$ (gold-, pred-).\\
$\text{Accuracy}=\dfrac{TP+TN}{TP+TN+FP+FN}$, $\text{Precision}=\dfrac{TP}{TP+FP}$, $\text{Recall}=\dfrac{TP}{TP+FN}$, $F_1=\dfrac{2PR}{P+R}$.\\
Macro: average per-class metrics equally; Micro: aggregate counts first, then compute.

\subsection*{A1 Repro (L3 Core)}
\[
\ell^{(i)}(\theta)= -\Big[y^{(i)}\log\sigma(\theta\cdot\phi(x^{(i)}))+(1-y^{(i)})\log\!\left(1-\sigma(\theta\cdot\phi(x^{(i)}))\right)\Big].
\]
\[
\nabla_\theta\ell^{(i)}=\left(\sigma(\theta\cdot\phi(x^{(i)}))-y^{(i)}\right)\phi(x^{(i)}).
\]

\subsection*{L3 Pitfalls}
- Accuracy can mislead under class imbalance; always report precision/recall/$F_1$.\\
- Too-large learning rate diverges; too-small learning rate stalls.\\
- Feature engineering/interactions often dominate final performance.
