\section{L3 Logistic Regression}
\subsection*{Model and Probabilities}
Data $(x^{(i)},y^{(i)})$, $y\in\{0,1\}$, score $z=w^\top x+b$, sigmoid $\sigma(z)=\dfrac{1}{1+e^{-z}}$.\\
\[
\hat{y}=\sigma(w^\top x+b),\ p(y=1|x)=\hat{y},\ p(y=0|x)=1-\hat{y}.
\]
Identity: $1-\sigma(a)=\sigma(-a)$; decision boundary at $w^\top x+b=0$ (threshold $0.5$).

\subsection*{Objective and Gradient}
Bernoulli form: $p(y\mid x)=\hat y^{\,y}(1-\hat y)^{1-y}$.\\
Cross-entropy/NLL: $L_{CE}(\hat y,y)=-\left[y\log\hat y+(1-y)\log(1-\hat y)\right]=-\log p(y\mid x;w,b)$.\\
For one sample: $\dfrac{\partial L}{\partial w_j}=(\hat y-y)x_j$, $\dfrac{\partial L}{\partial b}=\hat y-y$, vector form $\nabla_wL=(\hat y-y)x$.\\
SGD: $w\leftarrow w-\eta\nabla_wL$, $b\leftarrow b-\eta\dfrac{\partial L}{\partial b}$.

\subsection*{Evaluation}
$\text{Accuracy}=\dfrac{TP+TN}{TP+TN+FP+FN}$, $\text{Precision}=\dfrac{TP}{TP+FP}$, $\text{Recall}=\dfrac{TP}{TP+FN}$, $F_1=\dfrac{2PR}{P+R}$.\\
Macro: average per class; Micro: pool counts first.

\subsection*{A1 Check (L3)}
$\ell^{(i)}(\theta)=-\left[y^{(i)}\log\sigma(\theta\cdot\phi(x^{(i)}))+(1-y^{(i)})\log\left(1-\sigma(\theta\cdot\phi(x^{(i)}))\right)\right]$.\\
$\nabla_\theta\ell^{(i)}=\left(\sigma(\theta\cdot\phi(x^{(i)}))-y^{(i)}\right)\phi(x^{(i)})$.

\subsection*{L3 Reminders}
Class imbalance can make accuracy misleading; report precision/recall/$F_1$.\\
Learning rate too large diverges, too small stalls.
