\section{L2 N-gram + Perplexity}
\subsection*{Min Edit Distance}
For $X=x_1\ldots x_M$, $Y=y_1\ldots y_N$: $D(i,0)=i$, $D(0,j)=j$,\\
$D(i,j)=\min\{D(i-1,j)+1,\ D(i,j-1)+1,\ D(i-1,j-1)+cost\}$, with $cost=0$ if $x_i=y_j$.\\
Weighted form: use $del[x_i],\ ins[y_j],\ sub[x_i,y_j]$ in the same recurrence.\\
Backtrace stores DEL/INS/SUB-MATCH pointer; complexity $O(MN)$ time, $O(MN)$ (or rolling-row $O(\min\{M,N\})$) memory.

\subsection*{N-gram LM}
Chain rule: $P(w_1^n)=\prod_{i=1}^{n}P(w_i\mid w_1^{i-1})$; N-gram approx: $P(w_i\mid w_1^{i-1})\approx P(w_i\mid w_{i-N+1}^{i-1})$.\\
Bigram MLE: $P_{ML}(w_i\mid w_{i-1})=\dfrac{C(w_{i-1},w_i)}{C(w_{i-1})}$; unigram $P_{ML}(w_i)=\dfrac{C(w_i)}{\sum_{w\in V}C(w)}$.\\
Protocol: train for fitting, dev for tuning, test once for final reporting.

\subsection*{Perplexity}
$PP(W)=P(w_1\ldots w_N)^{-1/N}=\exp\!\left(-\dfrac{1}{N}\sum_{i=1}^{N}\log p_i\right)=2^{H(W)}$, $p_i=P(w_i\mid w_{<i})$.\\
Lower PP means higher normalized test probability (under same tokenizer/split).

\subsection*{Smoothing / Interpolation}
Add-1: $P_{add1}(w_i\mid w_{i-1})=\dfrac{C(w_{i-1},w_i)+1}{C(w_{i-1})+|V|}$; add-$k$: $\dfrac{C(w_{i-1},w_i)+k}{C(w_{i-1})+k|V|}$.\\
Trigram interpolation: $P=\lambda_3P_{ML}(w_i\mid w_{i-2},w_{i-1})+\lambda_2P_{ML}(w_i\mid w_{i-1})+\lambda_1P_{ML}(w_i)$, with $\lambda_1+\lambda_2+\lambda_3=1$.\\
Backoff: unseen higher-order context $\to$ lower-order model.

\subsection*{A1 Checks}
Q2(a) add-one bigram: $P(Sam\mid am)=\dfrac{3}{14}$; Q2(b) interpolation: $P(Sam\mid am)=\dfrac{31}{75}$.\\
Q3 without EOS: for fixed length $\ell$, $\sum_{w_{1:\ell}}P(w_{1:\ell})=1$, but not globally normalized across all lengths.

\subsection*{L2 Reminders}
Unsmoothed N-gram gives zero probability to unseen events; add-1 is often too aggressive for large $|V|$.\\
Tune interpolation weights on held-out data, not on the test set.
