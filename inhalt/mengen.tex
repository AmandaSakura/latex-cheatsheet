\section{L2: Min Edit Distance, N-grams, Perplexity, Smoothing}
\subsection*{Min Edit Distance (Levenshtein setup)}
Given strings $X=x_1\ldots x_M$, $Y=y_1\ldots y_N$:\\
\[
D(i,0)=i,\qquad D(0,j)=j.
\]
\[
\begin{array}{l}
D(i,j)=\min\{\\
\quad D(i-1,j)+1,\\
\quad D(i,j-1)+1,\\
\quad D(i-1,j-1)+cost\\
\}.
\end{array}
\]
with $cost=0$ if $x_i=y_j$, else substitution cost (often $2$ in lecture Levenshtein examples).\\
Weighted version:\\
\[
\begin{array}{l}
D(i,0)=D(i-1,0)+del[x_i],\\
D(0,j)=D(0,j-1)+ins[y_j].
\end{array}
\]
\[
\begin{array}{l}
D(i,j)=\min\{\\
\quad D(i-1,j)+del[x_i],\\
\quad D(i,j-1)+ins[y_j],\\
\quad D(i-1,j-1)+sub[x_i,y_j]\\
\}.
\end{array}
\]

\subsection*{Chain Rule and N-gram Approximation}
\[
P(w_1^n)=\prod_{i=1}^{n} P(w_i\mid w_1^{i-1}).
\]
\[
P(w_i\mid w_1^{i-1})\approx P(w_i\mid w_{i-(N-1)}^{i-1}).
\]
Markov approximation to last $N-1$ tokens.\\
MLE bigram:\\
\[
P_{ML}(w_i\mid w_{i-1})=\frac{C(w_{i-1},w_i)}{C(w_{i-1})}.
\]

\subsection*{Perplexity (canonical)}
For test sequence $W=w_1\ldots w_N$:\\
\[
\begin{array}{l}
PP(W)=P(w_1\ldots w_N)^{-1/N}\\
\qquad=\left(\prod_{i=1}^{N}P(w_i\mid w_1^{i-1})\right)^{-1/N}.
\end{array}
\]
\[
\begin{array}{l}
PP(W)=\exp\!\left(-\frac{1}{N}\sum_{i=1}^{N}\log\right.\\
\left.P(w_i\mid w_1^{i-1})\right).
\end{array}
\]
Equivalent base-2 form: $PP(W)=2^{H(W)}$.

\subsection*{Smoothing, Backoff, Interpolation}
Add-1 (Laplace) bigram:\\
\[
P_{add1}(w_i\mid w_{i-1})=\frac{C(w_{i-1},w_i)+1}{C(w_{i-1})+|V|}.
\]
Add-$k$ ($0<k<1$):\\
\[
P_{addk}(w_i\mid w_{i-1})=\frac{C(w_{i-1},w_i)+k}{C(w_{i-1})+k|V|}.
\]
Linear interpolation (trigram example):\\
\[
\begin{array}{l}
P(w_i\mid w_{i-2},w_{i-1})=\\
\lambda_3P_{ML}(w_i\mid w_{i-2},w_{i-1})+\\
\lambda_2P_{ML}(w_i\mid w_{i-1})+\lambda_1P_{ML}(w_i),
\end{array}
\]
$\lambda_1+\lambda_2+\lambda_3=1$.\\
Backoff idea: if higher-order count is zero, use lower-order model (discount/weight required; "stupid backoff" uses fixed scaling).

\subsection*{L2 Pitfalls}
- Unsmoothed model can assign zero probability to test set, making perplexity undefined/infinite.\\
- Add-1 is usually too blunt for large-vocabulary N-grams; interpolation/backoff is preferred.\\
- Compare perplexity only under the same tokenizer and evaluation protocol.
