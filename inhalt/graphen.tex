\section{L5+L6: Neural Nets, Transformer, LLM}
\subsection*{Feedforward Neural LM / Classifier}
Neural unit: $z=w\cdot x+b$, $a=f(z)$.\\
Activations:
\[
\sigma(z)=\frac{1}{1+e^{-z}}.
\]
\[
\tanh(z)=\frac{e^z-e^{-z}}{e^z+e^{-z}}.
\]
\[
\mathrm{ReLU}(z)=\max(0,z).
\]
Feedforward LM skeleton:
\[
x_t=[e_{w_{t-n+1}};\ldots;e_{w_{t-1}}].
\]
\[
h_t=g(Wx_t+b_1).
\]
\[
y_t=\mathrm{softmax}(Uh_t+b_2).
\]
\[
P(w_t\mid w_{<t})=y_t[w_t].
\]
\[
L_{CE}(t)=-\log y_t[w_{t+1}].
\]

\subsection*{L5 Multinomial/Softmax Regression}
Multinomial logistic regression (1-layer):
\[
y=\mathrm{softmax}(Wx+b).
\]
\[
y_c=p(y=c\mid x).
\]
\[
L=-\log p(y\mid x).
\]
Binary logistic is the $|C|=2$ special case.\\
Two-layer form: $h=g(Wx+b_1)$, $y=\mathrm{softmax}(Uh+b_2)$.

\subsection*{L5 Optional Regularization/Prior}
L2 regularization as MAP with Gaussian prior:
\[
\hat\theta=\arg\max_\theta\sum_i\log p(y^{(i)}\mid x^{(i)};\theta)-\lambda\|\theta\|_2^2.
\]
Interpretation: prior centered at $0$ encourages smaller weights.

\subsection*{Backprop / Chain Rule Mini-Template}
\[
\frac{d}{dx}u(v(x))=\frac{du}{dv}\frac{dv}{dx}.
\]
\[
\frac{d}{dx}u(v(w(x)))=\frac{du}{dv}\frac{dv}{dw}\frac{dw}{dx}.
\]
\[
\frac{d\sigma(z)}{dz}=\sigma(z)(1-\sigma(z)).
\]
\[
\frac{d\tanh(z)}{dz}=1-\tanh^2(z).
\]
\[
\frac{d\,\mathrm{ReLU}(z)}{dz}=
\begin{cases}
0,&z<0,\\
1,&z\ge 0.
\end{cases}
\]
SGD:
\[
\theta\leftarrow\theta-\eta\nabla_\theta L.
\]

\subsection*{Decoder-only LLM and Decoding}
Autoregressive factorization:
\[
P(w_1^n)=\prod_{i=1}^{n}P(w_i\mid w_{<i}).
\]
Greedy decoding:
\[
\hat w_t=\arg\max_{w\in V}P(w\mid w_{<t}).
\]
Logits to probs: $y=\mathrm{softmax}(u)$.\\
Temperature sampling: $\mathrm{softmax}(u/\tau)$, larger $\tau$ gives flatter distribution.\\
Quality/diversity tradeoff: flatter sampling increases diversity and lowers average quality.

\subsection*{Top-k and Top-p Decoding (L6)}
Top-$k$: keep $k$ highest-probability tokens, renormalize, then sample; $k=1$ is greedy.\\
Top-$p$ (nucleus): choose smallest set $S$ with $\sum_{w\in S}p(w)\ge p$, renormalize on $S$, sample.

\subsection*{Softmax + CE Quick Facts}
\[
y=\mathrm{softmax}(z).
\]
\[
y_i=\frac{e^{z_i}}{\sum_j e^{z_j}}.
\]
\[
L_{CE}(y,t)=-\sum_i t_i\log y_i.
\]
\[
\nabla_z L_{CE}=y-t.
\]
Temperature effect: $z/\tau$ with $\tau\downarrow$ sharper, $\tau\uparrow$ flatter.

\subsection*{Transformer Attention (canonical)}
Input encoding (decoder LM):
\[
x_i^0=E[w_i]+p_i.
\]
Single-head projections:
\[
q_i=x_iW^Q,\qquad k_j=x_jW^K.
\]
\[
v_j=x_jW^V.
\]
Scaled dot-product attention:
\[
score(x_i,x_j)=\frac{q_i\cdot k_j}{\sqrt{d_k}}.
\]
\[
\alpha_{ij}=\mathrm{softmax}(score(x_i,x_j)).
\]
\[
a_i=\sum_{j\le i}\alpha_{ij}v_j.
\]
Multi-head ($c=1,\ldots,h$):
\[
q_i^c=x_iW^{Q_c},\qquad k_j^c=x_jW^{K_c}.
\]
\[
v_j^c=x_jW^{V_c}.
\]
\[
head_i^c=\sum_{j\le i}\alpha_{ij}^c v_j^c.
\]
\[
a_i=(head_i^1\oplus\cdots\oplus head_i^h)W^O.
\]
Self-attention complexity (seq length $n$): time $O(n^2d)$, memory $O(n^2)$.

\subsection*{Causal Mask + LM Objective}
Causal (look-ahead) mask forbids attending to future positions:
\[
\alpha_{ij}=0\ \text{for}\ j>i.
\]
\[
\alpha_{ij}\propto \exp(score_{ij})\ \text{for}\ j\le i.
\]
Decoder LM training objective:
\[
L_{LM}=-\sum_{t=1}^{T}\log P(w_t\mid w_{<t}).
\]
Perplexity relation (same tokenizer/setup):
\[
PP=\exp\!\left(\frac{1}{T}L_{LM}\right).
\]

\subsection*{Sinusoidal Positional Encoding (if asked)}
\[
PE(pos,2i)=\sin\!\left(\frac{pos}{10000^{2i/d}}\right).
\]
\[
PE(pos,2i+1)=\cos\!\left(\frac{pos}{10000^{2i/d}}\right).
\]
Input is commonly $x_i^0=E[w_i]+PE_i$.

\subsection*{LayerNorm + Transformer Block (pre-norm)}
\[
\mu=\frac{1}{d}\sum_{r=1}^{d}x_r,
\qquad
\sigma=\sqrt{\frac{1}{d}\sum_{r=1}^{d}(x_r-\mu)^2}.
\]
\[
\mathrm{LayerNorm}(x)=\gamma\frac{x-\mu}{\sqrt{\sigma^2+\epsilon}}+\beta.
\]
Block equations:
\[
\begin{aligned}
t_i^1&=LN(x_i),\\
t_i^2&=MHA(t_i^1,\{x_1^1,\ldots,x_N^1\}),\\
t_i^3&=t_i^2+x_i,\\
t_i^4&=LN(t_i^3),\\
t_i^5&=FFN(t_i^4),\\
h_i&=t_i^5+t_i^3.
\end{aligned}
\]

\subsection*{Post-training Modes and Risks}
Continued pretraining: continue LM objective on domain corpus.\\
PEFT: tune small adapter/low-rank parameter subset.\\
SFT: supervised instruction-response fine-tuning.\\
Perplexity is tokenizer-sensitive; compare only under same tokenization and decoding setup.\\
Safety checklist: hallucination, privacy leakage, toxicity, representation bias.

\subsection*{A2-style Derivation Chain (transfer template)}
\textbf{Step 1:} write SGNS objective in positive + expected negative form:
\[
\begin{aligned}
J(\theta)={}&\sum_{(w,c)\in D}\log\sigma(u_c\cdot v_w)\\
&+k\sum_{(w,\cdot)\in D}\mathbb E_{c'\sim P_n}[g(w,c')],\\
g(w,c')={}&\log\sigma(-u_{c'}\cdot v_w),\\
&\max_\theta J(\theta).
\end{aligned}
\]
\textbf{Step 2:} expand expectation over vocabulary:
\[
\begin{aligned}
J(\theta)={}&\sum_{(w,c)\in D}\log\sigma(u_c\cdot v_w)\\
&+k\sum_{(w,\cdot)\in D}\sum_{c'\in V}P_n(c')g(w,c').
\end{aligned}
\]
\textbf{Step 3:} core derivative:
\[
\frac{d}{d\alpha}\log\sigma(\alpha)=\sigma(-\alpha)=\frac{1}{1+e^{\alpha}}.
\]
\textbf{Step 4:} stationary point for one pair ($\alpha=u_0\cdot v_1$):
\[
\alpha^*=\log\dfrac{A}{B}.
\]
\textbf{Step 5:} with unigram negative sampling:
\[
\alpha^*=PMI(1,0)-\log k.
\]
\textbf{Step 6:} choose $0<\beta<1$ in $P_n(w)\propto P_{unigram}(w)^\beta$; common choice $\beta=0.75$.

\subsection*{MCQ Transfer Triggers (L5/L6)}
If one option compares decoding quality/diversity, remember:\\
greedy/top-$k$ small $\Rightarrow$ quality up, diversity down;\\
temperature/top-$p$ larger $\Rightarrow$ diversity up.\\
If one option ignores tokenizer differences in perplexity comparison, reject it.\\
If one option allows future-token attention in decoder LM, reject it (breaks autoregressive setup).\\
If one option claims attention cost is linear in sequence length, reject it for vanilla self-attention.

\subsection*{L5/L6 Pitfalls}
- Most backprop bugs are wrong shape or wrong chain path.\\
- Wrong causal mask leaks future tokens and breaks autoregressive eval.\\
- Always report tokenizer + decoding setup with LM metrics.
