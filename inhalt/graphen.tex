\section{L5+L6: Neural Nets, Transformer, LLM, A2 Repro}
\subsection*{Feedforward Neural LM / Classifier Skeleton}
Neural unit:\\
\[
z=w\cdot x+b,\qquad a=f(z).
\]
Activation choices:\\
\[
\sigma(z)=\frac{1}{1+e^{-z}},\qquad
\tanh(z)=\frac{e^z-e^{-z}}{e^z+e^{-z}}.
\]
\[
\mathrm{ReLU}(z)=\max(0,z).
\]
Feedforward LM pattern:\\
\[
x_t=[e_{w_{t-n+1}};\ldots;e_{w_{t-1}}],
\]
\[
h_t=g(Wx_t+b_1).
\]
\[
y_t=\mathrm{softmax}(Uh_t+b_2).
\]
\[
P(w_t\mid w_{<t})=y_t[w_t].
\]
\[
L_{CE}(t)=-\log y_t[w_{t+1}].
\]

\subsection*{Backprop / Chain Rule Mini-Template}
\[
\frac{d}{dx}u(v(x))=\frac{du}{dv}\frac{dv}{dx}.
\]
\[
\frac{d}{dx}u(v(w(x)))=\frac{du}{dv}\frac{dv}{dw}\frac{dw}{dx}.
\]
Useful derivatives:\\
\[
\frac{d\sigma(z)}{dz}=\sigma(z)(1-\sigma(z)).
\]
\[
\frac{d\tanh(z)}{dz}=1-\tanh^2(z),
\]
\[
\frac{d\,\mathrm{ReLU}(z)}{dz}=\begin{cases}0,&z<0\\1,&z\ge 0\end{cases}
\]
SGD update: $\theta\leftarrow\theta-\eta\nabla_\theta L$.

\subsection*{Decoder-only LLM and Decoding}
Autoregressive factorization:\\
\[
P(w_1^n)=\prod_{i=1}^{n}P(w_i\mid w_{<i}).
\]
Greedy decoding:\\
\[
\hat w_t=\arg\max_{w\in V}P(w\mid w_{<t}).
\]
Logits $u$ to probabilities: $y=\mathrm{softmax}(u)$.

\subsection*{Transformer Attention (canonical)}
Single-head projections:\\
\[
q_i=x_iW^Q,\qquad k_j=x_jW^K.
\]
\[
v_j=x_jW^V.
\]
Scaled dot-product attention:\\
\[
score(x_i,x_j)=\frac{q_i\cdot k_j}{\sqrt{d_k}}.
\]
\[
\alpha_{ij}=\mathrm{softmax}(score(x_i,x_j)),
\]
\[
a_i=\sum_{j\le i}\alpha_{ij}v_j.
\]
Multi-head (head index $c=1..h$):\\
\[
q_i^c=x_iW^{Q_c},\qquad k_j^c=x_jW^{K_c}.
\]
\[
v_j^c=x_jW^{V_c},
\]
\[
head_i^c=\sum_{j\le i}\alpha_{ij}^c v_j^c.
\]
\[
a_i=(head_i^1\oplus\cdots\oplus head_i^h)W^O.
\]
LayerNorm (Eq. 9.21-9.24 form):\\
\[
\mu=\frac{1}{d}\sum_{r=1}^{d}x_r.
\]
\[
\sigma=\sqrt{\frac{1}{d}\sum_{r=1}^{d}(x_r-\mu)^2},
\]
\[
\mathrm{LayerNorm}(x)=\gamma\frac{x-\mu}{\sigma}+\beta.
\]
Transformer block (pre-norm residual stream, Eq. 9.25-9.30):\\
\[
t_i^1=\mathrm{LayerNorm}(x_i).
\]
\[
t_i^2=\mathrm{MHA}(t_i^1,\{x_1^1,\ldots,x_N^1\}),
\]
\[
t_i^3=t_i^2+x_i.
\]
\[
t_i^4=\mathrm{LayerNorm}(t_i^3),
\]
\[
t_i^5=\mathrm{FFN}(t_i^4),\qquad h_i=t_i^5+t_i^3.
\]

\subsection*{Post-training Modes and Risks}
Continued pretraining: continue LM objective on domain corpus.\\
PEFT: parameter-efficient tuning (e.g., adapter/low-rank style updates).\\
SFT: supervised fine-tuning on instruction-response pairs.\\
Perplexity caveat: tokenizer-sensitive; compare only with same tokenization.\\
Safety quick list: hallucination, privacy leakage, toxicity, representational bias.

\subsection*{A2 Repro (exam-speed)}
\textbf{Q1 MCQ key:} (a) C, (b) A, (c) B, (d) D, (e) A.\\
\textbf{Q2(a) SGNS objective:}\\
\[
\begin{array}{l}
\max_{\theta}\sum_{(w,c)\in\mathcal D}\log\sigma(u_c\cdot v_w)\\
\quad +k\sum_{(w,\cdot)\in\mathcal D}\mathbb E_{c'\sim P_n}\!\left[ \right.\\
\qquad \left.\log\sigma(-u_{c'}\cdot v_w)\right].
\end{array}
\]
Expanded expectation:\\
\[
\begin{array}{l}
\max_{\theta}\sum_{(w,c)\in\mathcal D}\log\sigma(u_c\cdot v_w)\\
\quad +k\sum_{(w,\cdot)\in\mathcal D}\sum_{c'\in V}P_n(c')\\
\qquad \cdot\log\sigma(-u_{c'}\cdot v_w).
\end{array}
\]
\textbf{Q2(b)}\\
\[
\frac{d}{d\alpha}\log\sigma(\alpha)=\frac{1}{1+e^{\alpha}}=\sigma(-\alpha).
\]
\textbf{Q2(c)} For one pair with $\alpha=u_0\cdot v_1$,\\
optimum: $\alpha^*=\log\dfrac{A}{B}$.\\
\textbf{Q2(d)} With unigram negative sampling, $\alpha^*=PMI(1,0)-\log k$.\\
\textbf{Q2(e)} Use $0<\beta<1$ in $P_n(w)\propto P_{unigram}(w)^\beta$; common choice $\beta=0.75$.

\subsection*{L5/L6 Pitfalls}
- Backprop errors usually come from wrong chain-rule path or shape mismatch.\\
- Attention masking/causal direction mistakes break autoregressive decoding.\\
- Report LM results with tokenizer and decoding setting; otherwise numbers are not comparable.
