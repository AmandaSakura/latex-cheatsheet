\section{L5+L6 Neural LM + Transformer}
\subsection*{Neural Units and Classifiers}
Neuron: $z=w\cdot x+b$, $a=f(z)$, with $\sigma(z)=\dfrac{1}{1+e^{-z}}$, $\tanh(z)=\dfrac{e^z-e^{-z}}{e^z+e^{-z}}$, $\mathrm{ReLU}(z)=\max(0,z)$.\\
Feedforward LM sketch: $x_t=[e_{w_{t-n+1}};\ldots;e_{w_{t-1}}]$, $h_t=g(Wx_t+b_1)$, $y_t=\mathrm{softmax}(Uh_t+b_2)$, $L_t=-\log y_t[w_{t+1}]$.\\
Multinomial logistic regression: $y=\mathrm{softmax}(Wx+b)$, binary logistic is the $|C|=2$ case.

\subsection*{Backprop / Optimization}
Chain rule: $\dfrac{d}{dx}u(v(x))=\dfrac{du}{dv}\dfrac{dv}{dx}$.\\
Derivatives: $\sigma'(z)=\sigma(z)(1-\sigma(z))$, $\tanh'(z)=1-\tanh^2(z)$, $\mathrm{ReLU}'(z)=\mathbf{1}[z\ge0]$.\\
SGD update: $\theta\leftarrow\theta-\eta\nabla_\theta L$; L2 regularization is MAP with Gaussian prior.

\subsection*{Autoregressive LM and Decoding}
Factorization: $P(w_1^n)=\prod_{i=1}^{n}P(w_i\mid w_{<i})$.\\
Greedy: $\hat w_t=\arg\max_w P(w\mid w_{<t})$; temperature: $\mathrm{softmax}(u/\tau)$ ($\tau\downarrow0$ more argmax-like, larger $\tau$ flatter).\\
Top-$k$: keep best $k$ tokens then sample (when $k=1$, equals greedy).\\
Top-$p$: keep smallest set $S$ with $\sum_{w\in S}p(w)\ge p_0$, renormalize, then sample.

\subsection*{Transformer Attention}
Single-head: $q_i=x_iW^Q$, $k_j=x_jW^K$, $v_j=x_jW^V$, score $=\dfrac{q_i\cdot k_j}{\sqrt{d_k}}$, $\alpha_{ij}=\mathrm{softmax}(score)$, $a_i=\sum_{j\le i}\alpha_{ij}v_j$.\\
Multi-head: compute heads in parallel and concatenate, then project by $W^O$.\\
LayerNorm: $\mu=\dfrac{1}{d}\sum_r x_r$, $\sigma=\sqrt{\dfrac{1}{d}\sum_r(x_r-\mu)^2}$, $\mathrm{LN}(x)=\gamma\dfrac{x-\mu}{\sigma}+\beta$.\\
Decoder block (pre-norm residual): LN $\to$ MHA $+$ residual $\to$ LN $\to$ FFN $+$ residual.

\subsection*{L6 Training Pipeline / Risks}
Pipeline: pretraining $\to$ instruction tuning $\to$ preference alignment.\\
Teacher forcing: train next-token CE with gold previous token as context.\\
PEFT tunes a small parameter subset; continued pretraining adapts domain.\\
Common risks: hallucination, privacy leakage, toxicity, representation bias.

\subsection*{A2 Checks}
SGNS objective: $J(\theta)=\sum_{(w,c)\in D}\log\sigma(u_c\cdot v_w)+k\sum_{(w,\cdot)\in D}\mathbb E_{c'\sim P_n}[\log\sigma(-u_{c'}\cdot v_w)]$.\\
$\dfrac{d}{d\alpha}\log\sigma(\alpha)=\sigma(-\alpha)$; optimal pair score in assignment derivation: $\alpha^*=\log\dfrac{A}{B}$, and with negative sampling $\alpha^*=PMI-\log k$.\\
Negative distribution: $P_n(w)\propto P_{unigram}(w)^\beta$, typically $0<\beta<1$ (commonly $0.75$).

\subsection*{L5/L6 Reminders}
Report tokenizer + decoding settings with perplexity or generation results.\\
Wrong causal mask leaks future tokens and invalidates autoregressive evaluation.
