\section{L5+L6: Neural Nets, Transformer, LLM, A2 Repro}
\subsection*{Feedforward Neural LM / Classifier}
Neural unit:
\[
z=w\cdot x+b,\qquad a=f(z).
\]
Activation choices:
\[
\sigma(z)=\frac{1}{1+e^{-z}},\qquad
\tanh(z)=\frac{e^z-e^{-z}}{e^z+e^{-z}},\qquad
\mathrm{ReLU}(z)=\max(0,z).
\]
XOR lesson: one linear unit cannot represent XOR; need hidden layer/nonlinearity.\\
Embedding matrix view (L5): token id $t$ maps to embedding $E_t$ from matrix $E\in\mathbb R^{|V|\times d}$.\\
Variable-length text classification commonly pools embeddings:
\[
h_{doc}=\mathrm{pool}(e_{w_1},\ldots,e_{w_T})\ \ (\text{mean/max}).
\]
Feedforward LM skeleton:
\[
x_t=[e_{w_{t-n+1}};\ldots;e_{w_{t-1}}],
\quad
h_t=g(Wx_t+b_1),
\]
\[
y_t=\mathrm{softmax}(Uh_t+b_2),\qquad
P(w_t\mid w_{t-n+1:t-1})=y_t[w_t],
\]
\[
L_{CE}(t)=-\log y_t[w_t].
\]

\subsection*{L5 Multinomial/Softmax Regression}
Lecture framing: multinomial logistic regression is a 1-layer network.\\
Input $x\in\mathbb R^d$, output class-probability vector $y\in\mathbb R^{|C|}$:
\[
y=\mathrm{softmax}(Wx+b),\qquad y_c=p(y=c\mid x).
\]
Multiclass CE/NLL: $L=-\log p(y\mid x)$; binary logistic is the $|C|=2$ special case.\\
Two-layer softmax classifier (slide form): $h=g(Wx+b_1)$, $y=\mathrm{softmax}(Uh+b_2)$.

\subsection*{L5 Optional Regularization/Prior}
If slide uses Gaussian-prior view: L2 regularization is MAP with Gaussian prior on weights.\\
Equivalent objective form:
\[
\hat\theta=\arg\max_\theta\ \sum_i \log p(y^{(i)}\mid x^{(i)};\theta)-\lambda\|\theta\|_2^2.
\]
Interpretation: prior centered at $0$ encourages smaller weights.

\subsection*{Backprop / Computation Graph Mini-Template}
\[
\frac{d}{dx}u(v(x))=\frac{du}{dv}\frac{dv}{dx},\qquad
\frac{d}{dx}u(v(w(x)))=\frac{du}{dv}\frac{dv}{dw}\frac{dw}{dx}.
\]
Derivatives used in lecture:
\[
\frac{d\sigma(z)}{dz}=\sigma(z)(1-\sigma(z)),\qquad
\frac{d\tanh(z)}{dz}=1-\tanh^2(z),
\]
\[
\frac{d\,\mathrm{ReLU}(z)}{dz}=
\begin{cases}
0,&z<0\\
1,&z\ge 0
\end{cases}
\]
SGD update:
\[
\theta\leftarrow\theta-\eta\nabla_\theta L.
\]
Implementation note: computation graph + backward pass gives gradients by chain rule.

\subsection*{Decoder-only LLM, Prompting, Decoding}
Autoregressive factorization:
\[
P(w_1^n)=\prod_{i=1}^{n}P(w_i\mid w_{<i}).
\]
Prompting: task is provided as text prefix; system prompt adds global behavior constraints.\\
Greedy decoding:
\[
\hat w_t=\arg\max_{w\in V}P(w\mid w_{<t}).
\]
Logits to probabilities:
\[
y=\mathrm{softmax}(u).
\]
Temperature sampling:
\[
\mathrm{softmax}(u/\tau),\qquad
\tau\uparrow \Rightarrow \text{flatter distribution}.
\]
Random sampling caveat: tail words are individually low-probability but have large total mass.\\
Quality/diversity tradeoff: sharper distribution $\Rightarrow$ quality up, diversity down.

\subsection*{Top-k and Top-p Decoding (L6)}
Top-$k$: keep only $k$ highest-probability tokens, renormalize, then sample.\\
When $k=1$, top-$k$ reduces to greedy decoding.\\
Top-$p$ (nucleus): choose smallest token set $S$ with cumulative probability $\sum_{w\in S}p(w)\ge p$, renormalize on $S$, then sample.\\
Why top-$p$: unlike fixed $k$, retained set size adapts to distribution sharpness by context.

\subsection*{Transformer Attention (canonical)}
Single-head projections:
\[
q_i=x_iW^Q,\qquad k_j=x_jW^K,\qquad v_j=x_jW^V.
\]
Masked scaled dot-product attention:
\[
\alpha_{ij}=\mathrm{softmax}\!\left(\frac{q_i\cdot k_j+m_{ij}}{\sqrt{d_k}}\right),
\qquad
m_{ij}=
\begin{cases}
0,&j\le i\\
-\infty,&j>i
\end{cases}
\]
\[
a_i=\sum_{j\le i}\alpha_{ij}v_j.
\]
Multi-head (head index $c=1,\ldots,h$):
\[
q_i^c=x_iW^{Q_c},\quad
k_j^c=x_jW^{K_c},\quad
v_j^c=x_jW^{V_c},
\]
\[
head_i^c=\sum_{j\le i}\alpha_{ij}^c v_j^c,\qquad
a_i=(head_i^1\oplus\cdots\oplus head_i^h)W^O.
\]
LayerNorm:
\[
\mu=\frac{1}{d}\sum_{r=1}^{d}x_r,\qquad
\sigma=\sqrt{\frac{1}{d}\sum_{r=1}^{d}(x_r-\mu)^2},
\]
\[
\mathrm{LayerNorm}(x)=\gamma\frac{x-\mu}{\sigma}+\beta.
\]
Transformer block (pre-norm residual stream):
\[
t_i^1=LN(x_i),\quad
t_i^2=MHA(t_i^1,\{x_1^1,\ldots,x_N^1\}),
\]
\[
t_i^3=t_i^2+x_i,\quad
t_i^4=LN(t_i^3),\quad
t_i^5=FFN(t_i^4),\quad
h_i=t_i^5+t_i^3.
\]

\subsection*{Training/Adaptation Modes and Risks}
Teacher forcing for LM training: at step $t$, model conditions on gold prefix $w_{<t}$.\\
Equivalent LM objective:
\[
L_{CE}=-\sum_t \log P(w_t\mid w_{<t}).
\]
Continued pretraining: continue LM objective on domain corpus.\\
PEFT: tune small adapter/low-rank parameter subset.\\
SFT: supervised instruction-response fine-tuning.\\
Perplexity caveat: tokenizer-sensitive; compare only under same tokenization.\\
Safety quick list: hallucination, privacy leakage, toxicity, representation bias.

\subsection*{A2 Repro (exam-speed chain)}
\textbf{Q1 MCQ key:} (a) C, (b) A, (c) B, (d) D, (e) A.\\
\textbf{Q2(a)} SGNS objective:
\[
\begin{array}{l}
J(\theta)=
\sum_{(w,c)\in D}\log\sigma(u_c\cdot v_w)\\
\quad +k\sum_{(w,\cdot)\in D}\mathbb E_{c'\sim P_n}[g(w,c')],
\end{array}
\]
\[
g(w,c')=\log\sigma(-u_{c'}\cdot v_w),\qquad
\max_\theta J(\theta).
\]
Expanded expectation form:
\[
\begin{array}{l}
J(\theta)=
\sum_{(w,c)\in D}\log\sigma(u_c\cdot v_w)\\
\quad +k\sum_{(w,\cdot)\in D}\sum_{c'\in V}P_n(c')\,g(w,c').
\end{array}
\]
\textbf{Q2(b)}
\[
\frac{d}{d\alpha}\log\sigma(\alpha)=\frac{1}{1+e^{\alpha}}=\sigma(-\alpha).
\]
\textbf{Q2(c)} for one pair ($\alpha=u_a\cdot v_b$):
\[
\alpha^*=\log\frac{A}{B}.
\]
\textbf{Q2(d)} with unigram negative sampling:
\[
\alpha^*=PMI(b,a)-\log k.
\]
\textbf{Q2(e)} choose $0<\beta<1$ in
\[
P_n(w)\propto P_{unigram}(w)^\beta,
\]
common choice $\beta=0.75$.
